# Stratify Payload_mass into four levels based on quantiles
labels = ['1 low     ', '2 mid-low ', '3 mid-high', '4 high    ']
df_link['Strata'] = pd.qcut(df_link['Payload_mass'], q=4, labels=labels)

# Get the quantile cuts without custom labels
df_link['Intervals'] = pd.qcut(df_link['Payload_mass'], q=4)

# Get unique intervals and their bounds
interval_bounds = {}
for interval in df_link['Intervals'].cat.categories:
    interval_bounds[str(interval)] = [interval.left, interval.right]

# Print out the bounds for each strata
for label, interval_str in zip(labels, interval_bounds.keys()):
    print(f"{label}: {interval_bounds[interval_str]}")

# Group by Orbit and Strata, then calculate the success rate
result = df_link.groupby(['Orbit', 'Strata']).Booster_landing.mean().reset_index()

# Get all combinations of Orbit and Strata
all_combinations = pd.MultiIndex.from_product([df_link['Orbit'].unique(), labels], 
                                              names=['Orbit', 'Strata']).to_frame(index=False)

# Left-join all combinations with the result dataframe
final_result = all_combinations.merge(result, on=['Orbit', 'Strata'], how='left')

# Reorder the columns to match your expected output
final_result = final_result[['Strata', 'Orbit', 'Booster_landing']]
final_result = final_result.rename(columns={'Booster_landing': 'success_rate'})
final_result = final_result.sort_values(by=['Strata', 'Orbit'])
final_result = final_result.reset_index(drop=True)
print(final_result)

plot = sns.barplot(x='Orbit', y='Booster_landing', data=success_rate)

for p in plot.patches:
    height = p.get_height()
    plot.annotate(f'{height:.2%}', (p.get_x() + p.get_width() / 2., p.get_height()),
                 ha='center', va='center', fontsize=10, color='black', xytext=(0, 5),
                 textcoords='offset points')

plt.title('Success Rate by Orbit')
plt.ylabel('Success Rate')
plt.xlabel('Orbit')
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()
