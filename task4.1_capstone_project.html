def task1_2(site):
    '''
    Crawl data from the given SpaceX API
    '''
    # Crawl data from the API
    crawl = requests.get(site)
    # Code of 200 means data retrieval is successful
    if crawl.status_code == 200:
        # Convert crawled data to JSON format
        data_api = crawl.json()
        # Convert data format from JSON to pandas DataFrame
        data_api = pd.DataFrame(data_api)
        # print(data_api.head())
    else:
        print(f"Data retrieval fails. Status code: {crawl.status_code}")
    return data_api

def polyline_dist(site_lat, site_lon, prox_lat, prox_lon):
    """
    site_lat: latitutde of launch site
    site_lon: longitude of launch site
    prox_lat: latitutde of a proximity
    prox_lon: longitude of a proximity
    
    Link a proximity to a launch site by a polyline, with distance calculated
    """
    # Calculate the distance between site and proximity
    R = 6371.0 # earth radius in km
    lat1 = radians(site_lat)
    lon1 = radians(site_lon)
    lat2 = radians(prox_lat)
    lon2 = radians(prox_lon)
    dlon = lon2 - lon1
    dlat = lat2 - lat1
    a = sin(dlat / 2)**2 + cos(lat1) * cos(lat2) * sin(dlon / 2)**2
    c = 2 * atan2(sqrt(a), sqrt(1 - a))
    dist = round(R * c, 4)
    
    # Add the distance marker showing the distance 
    distance_marker = folium.Marker(
        [prox_lat, prox_lon],
        icon=DivIcon(
            icon_size=(20,20),
            icon_anchor=(0,0),
            html=f'<div style="font-size: 12; color:#d35400; white-space: nowrap;"><b>{dist} km</b></div>'
            )
        )
    map.add_child(distance_marker)
    
    # Link the launch site and the proximities (e.g. railway)
    site = [site_lat, site_lon]
    prox = [prox_lat, prox_lon]
    line = folium.PolyLine(locations=[site, prox], weight=2)
    map.add_child(line)
    
    return map
