def task1_2(site):
    '''
    Crawl data from the given SpaceX API
    '''
    # Crawl data from the API
    crawl = requests.get(site)
    # Code of 200 means data retrieval is successful
    if crawl.status_code == 200:
        # Convert crawled data to JSON format
        data_api = crawl.json()
        # Convert data format from JSON to pandas DataFrame
        data_api = pd.DataFrame(data_api)
        # print(data_api.head())
    else:
        print(f"Data retrieval fails. Status code: {crawl.status_code}")
    return data_api

map = folium.Map(location=[site_lat_lon['lat'].mean(), site_lat_lon['lon'].mean()], zoom_start=4)

site_list = [[site_lat_lon.iloc[i,1], site_lat_lon.iloc[i,2]] for i in range(3)]
colors = ['purple', 'red', 'green']

for i in range(3):
  folium.CircleMarker(
    location=site_list[i],
    radius=50, # in pixels
    color=colors[i],
    fill=True,
    fill_color=colors[i],
    tooltip=folium.Tooltip(f"{site_lat_lon.iloc[i,0]} ({colors[i]})")
    #popup=folium.Popup(site_lat_lon.iloc[i,0])
    ).add_to(map)

marker_cluster = MarkerCluster().add_to(map)
for _, row in df_link.iterrows():
    color = 'green' if row['Booster_landing'] == 1 else 'red'
    folium.Marker(
        location=site_list[site.index(row['Launch_site'])],
        icon=folium.Icon(color=color),
        popup=row['Booster_landing']
    ).add_to(marker_cluster)
